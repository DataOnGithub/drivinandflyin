define([
        "./Popup",
        "./DomUtil",
        "./Utils",
        "./i18n"
    ],
    
    function(base, du, utils) {
		
		tomtom.TrafficIncidentPopup = OpenLayers.Class(base, {
			
			/**
			 * A popup which is displayed when a user clicks on a traffic incident icon.
			 * @constructor
			 * @class TrafficIncidentPopup
			 * @namespace tomtom
			 * @extends tomtom.Popup
			 * @param {Object} incident The traffic incident for which information is displayed.
			 * @param {tomtom.TrafficMarker} marker The traffic marker for which the popup is being displayed.
			 */
			initialize: function(incident, marker) {
				
				var baseUrl = OpenLayers.ImgPath;
				var bundle = new tomtom.StringBundle("IncidentItem");
				
				// default properties to empty
				var data = du.mixin({
					r: "", 
					dl: "--", 
					l: "--", 
					d: "", 
					f: "--", 
					t: "--",
					icon: "url(" + baseUrl + "traffic-icons/" + marker.getCategoryString(incident.ic) + "-" + marker.getTypeString(incident.ty) + ".png)",
					"labelDelay": bundle.getString("delay"),
					"labelLength": bundle.getString("length"),
					"labelFrom": bundle.getString("from"),
					"labelTo": bundle.getString("to")
				}, incident);
				
				// format the delay time
				if (data.dl && !isNaN(data.dl)) {
					data.dl = Math.round(data.dl / 60);
					data.dl += " " + tomtom.StringBundle.getString("units.time.minute", null, data.dl);
				}
				
				// format the distance
				if (data.l && !isNaN(data.l))
					data.l = tomtom.StringBundle.getString("units.distance", null, data.l);
				
				// build the markup for the popup
				var markup = OpenLayers.String.format(TEMPLATE, data);
				
				// show the popup
				base.prototype.initialize.apply(this, [{
					id: "traffic_popup_" + incident.id, 
					marker: marker, 
					contentSize: new OpenLayers.Size(325, 150), 
					contentHTML: markup
				}]);
				
			}
			
		});
		
		var TEMPLATE = "<div class=\"trafficInfo\">" +
				"<div class=\"header\">" +
				"<div style=\"background-image: ${icon}\" class=\"icon\"></div>" +
				"<div class=\"road\">${r}</div>" +
				"<div class=\"delay\"><label>${labelDelay}:</label>${dl}</div>" +
				"<div class=\"length\"><label>${labelLength}:</label>${l}</div>" +
				"</div>" +
				"<div class=\"body\">" +
				"<div class=\"description\">${d}</div>" +
				"<div class=\"location\">" +
				"<div class=\"line\"><label>${labelFrom}:</label><div>${f}</div></div>" +
				"<div class=\"line\"><label>${labelTo}:</label><div>${t}</div></div>" +
				"</div></div></div>";
		
		return tomtom.TrafficIncidentPopup;
	
	}
);
