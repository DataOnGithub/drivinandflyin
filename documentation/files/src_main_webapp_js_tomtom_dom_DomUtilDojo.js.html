/**
 * jQuery implementation of DomUtil.
 */
define([
		"./dom/DomUtil",
		"dojo/dom-style",
		"dojo/query",
		"dojo/_base/fx",
		"dojo/dom-geometry",
		"dojo/fx/easing",
		"dojo/_base/connect",
		"./dom/DomUtilResult",
		"./Utils",
		"dojo/NodeList-dom",
		"dojo/io/script",
		"dojo/_base/lang",
		"dojo/dom-class",
		"dojo/dom-attr",
		"dojo/NodeList-data",
		"dojo/NodeList-manipulate",
		"dojo/NodeList-traverse",
		"dojo/_base/array",
		"dojo/window",
		"dojo/_base/xhr"
	], 
	
	function(dutil, style, dq, fx, domgeo, fxeasing, 
			connect, result, util, ddom, script, lang, 
			domClass, domAttr, nodeListData, manip, traverse, array, win, xhr) {
	
		tomtom.dom.DomUtilDojo = OpenLayers.Class(dutil, {
			
			initialize: function() { },
			
			query: function(query, baseElement) {
				
				if (baseElement && (this._isNodeList(baseElement) || baseElement.query)) {					
					return new result(baseElement.query(query), this);
				} else {
					return new result(dq(query, baseElement), this);
				}
				
			},
			
			animate: function(element, options) {
				if (!options)
					options = {};
				
				var easing = null;
				
				if (options.easing) {
					easing = tomtom.dom.DomUtilDojo.getEasingFunction(options.easing);
				}
				
				fx.animateProperty({
					node: element,
					properties: options.properties,
					easing: easing,
					duration: options.duration,
					onEnd: options.callback
				}).play();
				
				return new result(element, this);
			},
			
			offset: function(element) {
				var box = domgeo.position(element);
				
				return {
					top: box.x,
					left: box.y
				};
			},
			
			dimensions: function(element) {
				var box = domgeo.getMarginBox(element);
				var position = domgeo.position(element);
				
				return {
					width: position.w,
					height: position.h,
					outerWidth: box.w,
					outerHeight: box.h
				};
			},
			
			position: function(element) {
				var box = domgeo.getMarginBox(element);
				
				return {
					top: box.t,
					left: box.l
				};
			},
			
			css: function(element, property, value) {
				
				if (value != null) {
					style.set(element, property, value);
					return new result(element, this);
				} else
					return style.get(element, property);
			},
			
			jsonp: function(options) {
				script.get({
					url: options.url,
					callbackParamName: options.callbackParam,
					content: options.data,
					load: options.callback,
					error: options.errorCallback
				});
			},
			
			getJson: function(options) {
				xhr.get({
			        url:options.url,
			        handleAs:"json",
			        load: options.callback,
			        content: options.data,
			        error: options.errorCallback
			    });
			},
			
			bind: function(element, event, callback, scope) {
				if (this._isNodeList(element) || element.on) {
					return element.on(event, scope ? dojo.hitch(scope, callback) : callback);
				} else {
					return connect.connect(element, "on" + event, element, scope ? dojo.hitch(scope, callback) : callback);
				}
			},
			
			unbind: function(handle) {
				connect.disconnect(handle);
			},
			
			mixin: function(object, mixin) {
				return lang.mixin(object, mixin);
			},
			
			addClass: function(element, className) {
				if (this._isNodeList(element) || util.isArray(element)) {
					array.forEach(element, function(elem) {
						domClass.add(elem, className);
					});
				} else {
					domClass.add(element, className);
				}
			},
			
			removeClass: function(element, className) {
				if (this._isNodeList(element) || util.isArray(element)) {
					array.forEach(element, function(elem) {
						domClass.remove(elem, className);
					});
				} else {
					domClass.remove(element, className);
				}
			},
			
			attr: function(element, name, value) {
				if (value != null) {
					domAttr.set(element, name, value);
					return new result(element, this);
				} else
					return domAttr.get(element, name);
			},
			
			removeAttr: function(element, name) {
				domAttr.remove(element, name);
				return new result(element, this);
			},
			
			data: function(element, name, value) {
				if (value !== undefined) {
					nodeListData(element).data(name, value);
					return new result(element, this);
				} else
					return nodeListData(element).data(name);
			},
			
			val: function(element, value) {				
				if (value != undefined) {
					manip(element).val(value);
					return new result(element, this);
				} else
					return manip(element).val();
			},
			
			next: function(element, filter) {
				return new result(traverse(element).next(filter), this);
			},
			
			prev: function(element, filter) {
				return new result(traverse(element).prev(filter), this);
			},
			
			children: function(element, filter) {
				return new result(traverse(element).childen(filter), this);
			},
			
			viewport: function() {
				var box = win.getBox();
				
				return {
					width: box.w,
					height: box.h
				};
			},
			
			_isNodeList: function(obj) {
				return Object.prototype.toString.call(obj) === "[object NodeList]";
			}
			
		});
		
		tomtom.dom.DomUtilDojo.getEasingFunction = function(name) {
			var translatedName = "";
			// translate the easing function names
			
			switch(name) {
			case "easeOutBounce":
				translatedName = "bounceOut";
				break;
			case "easeInBounce":
				translatedName = "bounceIn";
				break;
			case "easeOutCirc":
				translatedName = "circOut";
				break;
			}
			
			return fxeasing[translatedName];
		};
		
		var DEFAULT_ANIMATION_DURATION = 500;
		
		return new tomtom.dom.DomUtilDojo();
	}
);
