define([
        "./Main"
    ],
    
    function(base) {
	
		var base = OpenLayers.Icon;
		
		tomtom.CustomIcon = OpenLayers.Class(base, {
			
			/**
			 * A custom icon implementation that allows for CSS customization.
			 * This class is primarily used by {{#crossLink "tomtom.CustomMarker"}}CustomMarker{{/crossLink}}.  However, it can be used
			 * in any custom marker type of implementation.
			 * @constructor
			 * @class CustomIcon
			 * @extends OpenLayers.Icon
			 * @param {String} className The CSS class the icon should have.
			 */
			initialize: function(className) {

				this.url = "";
				this.size = new OpenLayers.Size(0, 0);
				this.calculateOffset = false;
				this.imageDiv = document.createElement("div");
				this.imageDiv.id = OpenLayers.Util.createUniqueID("OL_Icon_");
				this.imageDiv.className = className;
				
			},

			/**
			 * Creates a copy of this instance.
			 * @method clone
			 * @returns {tomtom.CustomIcon}
			 */
		    clone: function() {
		        return new tomtom.CustomIcon(this.imageDiv.className);
		    },
		    
		    /**
		     * Destroys this icon making sure to release any resources it is holding on to.
			 * @method destroy
		     */
		    destroy: function() {
		    	
		    	this.erase();
		    	
		    	this.imageDiv.innerHTML = "";
		    	this.imageDiv = null;
		    	
		    },
		    
		    draw: function(px) {
		        this.moveTo(px);
		        return this.imageDiv;
		    }, 
		    
		    setOpacity: function(opacity) {
		    	OpenLayers.Util.modifyDOMElement(this.imageDiv, null, null, null, null, null, null, opacity);
		    },
		    
		    moveTo: function (px) {
		        //if no px passed in, use stored location
		        if (px != null) {
		            this.px = px;
		        }
		
		        if (this.imageDiv != null) {
		            if (this.px == null) {
		                this.display(false);
		            } else {
		            	OpenLayers.Util.modifyDOMElement(this.imageDiv, null, {
		                    x: this.px.x + (this.offset ? this.offset.x : 0),
		                    y: this.px.y + (this.offset ? this.offset.y : 0)
		                });
		            }
		        }
		    }
			
		});
		
		return tomtom.CustomIcon;
	
	}
);
