define([
		"./CustomMarker"
	], 
	function(base) {
	
		tomtom.ClusterMarker = OpenLayers.Class(base, {
			
			/**
			 * A marker which is used to group multiple markers into a cluster.
			 * This is primarily used by the {{#crossLink "MarkerManager"}}MarkerManager{{/crossLink}} class.
			 * @constructor
			 * @class ClusterMarker
			 * @extends tomtom.Marker
			 * @param {tomtom.Cluster} cluster The cluster for which the marker is being created.
			 */
			initialize: function(cluster) {
				/**
				 * The cluster for which the marker was created.
				 * @property cluster
				 */
				this.cluster = cluster;
				
				var lonlat = cluster.baseMarker.lonlat;
				
				base.prototype.initialize.apply(this, [lonlat.lon, lonlat.lat, "olMarkerCluster"]);
				
				this.update();
				
			},
			
			/**
			 * Updates the displayed count on the cluster based on the number of markers in the cluster.
			 * @method update
			 */
			update: function() {
				// set the number of markers in the cluster
				this.icon.imageDiv.innerHTML = this.cluster.markers.length;
			}
			
		});
		
		return tomtom.ClusterMarker;
		
	}
);
