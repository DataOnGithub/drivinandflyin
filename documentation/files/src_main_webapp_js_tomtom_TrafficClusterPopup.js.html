define([
        "./Popup",
        "./DomUtil",
        "./Utils",
        "./TrafficIncidentPopup"
    ],
    
    function(base, du, utils) {
		
		tomtom.TrafficClusterPopup = OpenLayers.Class(base, {
			
			/**
			 * A popup which is used to display information about a traffic cluster.
			 * @constructor
			 * @class TrafficClusterPopup
			 * @namespace tomtom
			 * @extends tomtom.Popup
			 * @param {Object} incident The incident information returned from the traffic service.
			 * @param {tomtom.TrafficIcon} marker The traffic marker the popup is displayed for.
			 */
			initialize: function(incident, marker) {

				this.incident = incident;
				this.marker = marker;
				
				var baseUrl = OpenLayers.ImgPath;
				var poiCount = 0;
				var poiMarkup = "";
				var bundle = new tomtom.StringBundle("IncidentBalloon.Cluster");
				
				// sort the POIs by the longest delay
				var pois = this.pois = incident.cpoi.sort(function(a, b) {
					if (a.dl > b.dl)
						return -1;
					else if (a.dl == b.dl)
						return 0;
					else
						return 1;
				});
				
				// generate the markup for the top 4 POIs
				for (var poiIndex = 0; poiIndex < pois.length && poiIndex < 4; poiIndex++) {
					var poi = pois[poiIndex];
					var className = "";
					
					// trim the from / to fields
					if (poi.f && poi.f.length > CLUSTER_ROAD_NAME_MAX_LENGTH)
						poi.f = poi.f.substring(0, CLUSTER_ROAD_NAME_MAX_LENGTH) + "...";
					
					if (poi.t && poi.t.length > CLUSTER_ROAD_NAME_MAX_LENGTH)
						poi.t = poi.t.substring(0, CLUSTER_ROAD_NAME_MAX_LENGTH) + "...";
					
					if (poiIndex + 1 == pois.length)
						className = "last";
					
					var data = du.mixin({
						r: "", 
						dl: "--", 
						l: "--", 
						d: "", 
						f: "--", 
						t: "--",
						index: poiIndex,
						className: className,
						icon: "url(" + baseUrl + "traffic-icons/" + marker.getCategoryString(poi.ic) + "-" + marker.getTypeString(poi.ty) + ".png)"
					}, poi);
					
					// format the delay time
					if (data.dl && !isNaN(data.dl)) {
						data.dl = Math.round(data.dl / 60);
						data.dl += " " + tomtom.StringBundle.getString("units.time.minute", null, data.dl);
					}
					
					// format the distance
					if (data.l && !isNaN(data.l))
						data.l = tomtom.StringBundle.getString("units.distance", null, data.l);
					
					poiMarkup += OpenLayers.String.format(INFO_BUBBLE_CLUSTER_ROW_TEMPLATE, data);
					
					poiCount++;
				}
				
				// generate the main markup
				var markup = OpenLayers.String.format(INFO_BUBBLE_CLUSTER_TEMPLATE, {
					incidentCount: incident.cs,
					labelIncidents: bundle.getString("title"),
					mostSevereIncidents: bundle.getString("listHeader", { count: poiCount }, poiCount),
					orderedBy: bundle.getString("orderedByDelay"),
					incidents: poiMarkup
				});
				
				// show the popup
				base.prototype.initialize.apply(this, [{ 
					id: "traffic_popup_" + incident.id, 
					marker: marker, 
					contentSize: new OpenLayers.Size(325, 225), 
					contentHTML: markup
				}]);
				
			},
			
			setMap: function(map) {
				
				// make sure and unbind the previous click handler
				if (this._rowClickHandle)
					du.unbind(this._rowClickHandle);
				
				var popup = this;
				
				// attach to the click event of the incident rows
				this._rowClickHandle = du.query("tr", this.div).bind("click", function(e) {
					var poiIndex = parseInt(du.attr(this, "data-poi-index"));
					var incident = null;
					
					if (!isNaN(poiIndex))
						incident = popup.pois[poiIndex];
					
					if (incident) {
						var newPopup = new tomtom.TrafficIncidentPopup(incident, popup.marker);
						map.addPopup(newPopup, true);
						newPopup.show();
					}
					
					// make sure the event doesn't bubble up, if it did the popup would close instantly
					OpenLayers.Event.stop(e);
				});
				
			},
			
			destroy: function() {
				
				if (this._rowClickHandle)
					du.unbind(this._rowClickHandle);
				
				base.prototype.destroy.apply(this, arguments);
				
			}
			
		});
		
		var INFO_BUBBLE_CLUSTER_TEMPLATE = "<div class=\"trafficInfo trafficInfo-cluster\">" +
				"<div class=\"header\">${labelIncidents} ${incidentCount}</div>" +
				"<h3>${mostSevereIncidents} <span>${orderedBy}</span></h3>" +
				"<table>${incidents}</table>" +
				"</div>";
		
		var INFO_BUBBLE_CLUSTER_ROW_TEMPLATE = "<tr class=\"${className}\" data-poi-index=\"${index}\">" +
				"<td><div class=\"icon\" style=\"background-image: ${icon}\"></div></td>" +
				"<td class=\"location\"><div class=\"from\">${f}</div><div class=\"to\">${t}</div></td>" +
				"<td class=\"delay\">${dl}</td>" +
				"<td class=\"length\">${l}</td>";
		
		var CLUSTER_ROAD_NAME_MAX_LENGTH = 18;
		
		return tomtom.TrafficClusterPopup;
	
	}
);
