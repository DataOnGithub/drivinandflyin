define([
		"../Main",
		"../OpenLayers"
	],
	
	function() {
	
		var base = OpenLayers.Layer.Markers;
		
		tomtom.layers.MarkerLayer = OpenLayers.Class(base, {
			
			/**
			 * Creates a marker layer which can be used to add markers to the map.
			 * A map can have multiple marker layers.  The map creates a "default" marker layer which is used when the {{#crossLink "tomtom.Map/addMarker"}}addMarker{{/crossLink}}
			 * method is called.
			 * 
			 * A marker layer can be added just like any other layer:
			 * 
			 * 	var layer = new tomtom.layers.MarkerLayer("MyMarkerLayer");
			 * 	map.addLayer(layer);
			 * 
			 * @constructor
			 * @class MarkerLayer
			 * @extends OpenLayers.Layer.Markers
			 * @namespace tomtom.layers
			 * @param {String} name The name of the marker layer.
			 */
			initialize: function() {
				
				base.prototype.initialize.apply(this, arguments);
				
			},
			
			/**
			 * Adds a marker to the layer.
			 * @method addMarker
			 * @param {tomtom.Marker} marker The marker to add.
			 * @param {boolean} [transform=true] Whether or not to transform the coordinates of the marker's lon / lat to EPSG:900913.
			 */
			addMarker: function(marker, transform) {
				
				if (typeof(transform) === "undefined")
					transform = true;
				
				if (transform) {
					// transform the lat / lon
					marker.lonlat.transform(
						this.projection,
						this.map.getProjectionObject()
					);
				}
				
				// make sure the marker knows it has been added
				if (marker.setMap)
					marker.setMap(this.map);
				
				base.prototype.addMarker.apply(this, arguments);
				
			}
			
			/**
			 * Removes a marker from the layer.
			 * @method removeMarker
			 * @param {tomtom.Marker} marker The marker to remove.
			 */
			
		});
		
		return tomtom.layers.MarkerLayer;
	
	}
);
