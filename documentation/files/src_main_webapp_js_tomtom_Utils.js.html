define(["./Main"], function() {
	
	tomtom.Utils = {
			
		hostIndex: 0,
		hosts: [],
		
		getServiceUrl: function(url, queryParams, cycleHosts, https) {
			
			var base = tomtom.API_BASE_URL;
			
			// add in the url parameters
			if (queryParams) {
				paramCount = 0;
				
				if (url.indexOf("?") == -1)
					url += "?";
				
				for (var param in queryParams) {
					if (paramCount > 0)
						url += "&";
					
					url += param + "=" + queryParams[param];
					
					paramCount++;
				}
			}
			
			// check to see if a proxy is being used
			if (tomtom.proxyUrl) {
				base = tomtom.proxyUrl;
			}
			
			// check to see if HTTPS should be enabled for this service
			if (typeof(https) != "undefined") {
				if (https) {
					base = base.replace(/https?:\/\//g, "https://");
				} else {
					base = base.replace(/https?:\/\//g, "http://");
				}
			}
			
			if (cycleHosts) {
				base = "http://" + this.getRandomHost(base);
			}
			
			return base + url;
			
		},
		
		getRandomHost: function(host) {
			host = host.replace(/https?:\/\//g, "");
			
			if (typeof(tomtom.enableHostCycling) == "undefined" || tomtom.enableHostCycling) {
				
				if (this.hosts.length == 0)
					this.initHosts();
				
				host = this.hosts[this.hostIndex];
			
				// increment the host index
		        this.hostIndex++;
		        
		        // if the host index has exceeded the max, go back to the start
		        if (this.hostIndex > 3)
		        	this.hostIndex = 0;
				
			}
			
	        return host;
	        
		},
		
		initHosts: function() {
			var url = tomtom.API_BASE_URL;
			
			// check to see if proxying is enabled
			if (tomtom.proxyUrl)
				url = tomtom.proxyUrl;
			
			tomtom.Utils.hosts.push(url.replace(/https?:\/\//g, "a."));
			tomtom.Utils.hosts.push(url.replace(/https?:\/\//g, "b."));
			tomtom.Utils.hosts.push(url.replace(/https?:\/\//g, "c."));
			tomtom.Utils.hosts.push(url.replace(/https?:\/\//g, "d."));
		},
		
		isArray: function(obj) {
			return Object.prototype.toString.call(obj) === '[object Array]';
		},
		
		/**
         * Zero-pad a positive integer.
         * number - {Int} 
         * length - {Int} 
         *
         * Returns:
         * {String} A zero-padded string
         */		
		zeroPad: function(number, length) {
            number = String(number);
            var zeros = [];
            for(var i=0; i<length; ++i) {
                zeros.push('0');
            }
            return zeros.join('').substring(0, length - number.length) + number;
        },
        
        hitch: function(scope, callback) {
        	return function() {
				return callback.apply(scope, arguments);
			};
        },
        
        setCookie: function(name,value,days) {
        	if (days) {
        		var date = new Date();
        		date.setTime(date.getTime()+(days*24*60*60*1000));
        		var expires = "; expires="+date.toGMTString();
        	}
        	else var expires = "";
        	document.cookie = name+"="+value+expires+"; path=/";
        },

        getCookie: function(name) {
        	var nameEQ = name + "=";
        	var ca = document.cookie.split(';');
        	for(var i=0;i < ca.length;i++) {
        		var c = ca[i];
        		while (c.charAt(0)==' ') c = c.substring(1,c.length);
        		if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
        	}
        	return null;
        },
        
        removeCookie: function(name) {
        	this.setCookie(name,"",-1);
        },
        
        formatSeconds: function(seconds) {
        	if (seconds == 0)
        		return "0 min";
        	else if (seconds < 60)
        		return seconds + " secs";
        	else if (seconds >= 60 && seconds <= 3600)
        		return this.round(seconds / 60, 1) + " mins";
        	else if (seconds >= 3600)
        		return this.round(seconds / 3600, 1) + " hours";
        	
        },
        
        metersToMiles: function(meters) {
        	
        	return meters * FEET_IN_METER / 5280;
        	
        },
        
        round: function(num, dec) {
        	return Math.round(num * Math.pow(10, dec)) / Math.pow(10, dec);
        }
		
	};
	
	var FEET_IN_METER = 3.28084;
	var FEET_IN_MILE = 5280;
	
	return tomtom.Utils;
	
});
