define([
		"./Popup",
        "text!templates/ClusterPopupItem.html",
        "./DomUtil",
        "./OpenLayers"
	],
	
	function(base, itemTemplate, du) {
	
		tomtom.ClusterPopup = OpenLayers.Class(base, {
			
			/**
			 * A popup class which displays information about a cluster of markers.
			 * This is the default popup that is displayed for clusters.
			 * @constructor
			 * @class ClusterPopup
			 * @extends tomtom.Popup
			 * @param {Object} options Additional options for the popup.  For more detailed information, see {{#crossLink "tomtom.Popup"}}Popup{{/crossLink}}.
			 * @param {tomtom.Cluster} options.cluster The cluster for which the popup is being created.
			 */
			initialize: function(options) {
				this._eventHandles = [];
				
				var markup = "<div class=\"cluster\">";
				var cluster = options.cluster;
				this.cluster = options.cluster;
				
				// add an item for each clustered marker
				for (var markerIndex = 0; markerIndex < cluster.markers.length; markerIndex++) {
					var marker = cluster.markers[markerIndex];
					
					var icon = marker.icon.url;
					var background = "url(" + icon + ");";
					
					markup += OpenLayers.String.format(itemTemplate, {
						background: background,
						description: marker.getData() ? marker.getData().description : "",
						markerIndex: markerIndex
					});
				}
				
				base.prototype.initialize.apply(this, [OpenLayers.Util.applyDefaults({
					contentHTML: markup + "</div>",
					marker: cluster.marker
				}, options)]);
			},
			
			show: function() {
				
				var me = this;
				
				this._eventHandles.push(du.query(".clusterItem", this.div).bind("click", function() {
					var marker = me.cluster.markers[du.attr(this, "data-markerIndex")];
					
					// create a new popup using the marker's popup options
					if (marker && marker.popup) {
						var options = marker.popupOptions;
						var map = me.cluster.marker.map;
						options.marker = me.cluster.marker;
						
						var popup = new marker.popupClass(options);
						map.addPopup(popup, true);
						popup.show();
						
						// wire up the hide event to destroy the popup
						popup.events.register("hide", this, function() {
							popup.destroy();
						});
						
						this.currentPopup = popup;
					}
				}));
				
				base.prototype.show.apply(this, arguments);
				
			},
			
			hide: function() {
				du.unbindAll(this._eventHandles);
				
				base.prototype.hide.apply(this, arguments);
			},
			
			destroy: function() {
				du.unbindAll(this._eventHandles);
				
				if (this.currentPopup && this.currentPopup.div)
					this.currentPopup.destroy();
				
				base.prototype.destroy.apply(this, arguments);
			}
			
		});
		
		return tomtom.ClusterPopup;
		
	}

);
